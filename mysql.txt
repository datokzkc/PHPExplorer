TABLE

CREATE TABLE tags_all(
tag_id TINYINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
tag VARCHAR(512) NOT NULL);

CREATE TABLE dirs_all(
dir_id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
path TEXT NOT NULL);

CREATE TABLE dir_tag(
id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
dir_id INT UNSIGNED NOT NULL,
tag_id TINYINT UNSIGNED NOT NULL,
FOREIGN KEY (tag_id)
REFERENCES tags_all(tag_id)
ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY (dir_id)
REFERENCES dirs_all(dir_id)
ON UPDATE CASCADE ON DELETE CASCADE);
